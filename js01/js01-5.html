불자료형

        javascript 서식                   //     결과
        console.log('!true : '+b1)        // !true : false
        console.log('!false : '+b2)       // !false : true
        console.log('!!false : '+b2)      // !!false : false → ! 불연산자 2번 사용
        console.log('57>333 : '+b3)       // 57>333 : false
        console.log("123 == '123' : "+b4) // 123 == '123' : true
        console.log('"사랑">"하마" : '+b5) // "사랑">"하마" : false
