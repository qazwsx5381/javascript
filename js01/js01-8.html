                    javascript 서식                        //    결과
        const result1 = 3+6
        console.log(`3 더하기 6은 ${result1} 입니다.`)      // 3 더하기 6은 9 입니다.
        console.log(`13 X 26 = ${13*26}입니다.`)           // 13 X 26 = 338입니다.
        console.log(`1 == '1' => ${1=='1'}`)              // 1 == '1' => true
        console.log(`false == 0 => ${false==0}`)          // false == 0 => true
        console.log(`false === 0 => ${false===0}`)        // false === 0 => false
        console.log(`true == '1' => ${true=='1'}`)        // true == '1' => true
        console.log(`"" == [] => ${""==[]}`)              // "" == [] => true
        console.log(`0 == [] => ${0 == []}`)              // 0 == [] => true
