상수와 변수, 복합대입연산자

const : 변하지 않는 값을 대입
선언 : const 이름 = 값
let : 변할 가능성이 있는 값을 대입
선언 : let 이름 = 값

        javascript 서식                                          // 결과
        console.log(`a=${a}, b=${b}, c=${c}, d=${d}, e=${e}`)    // a=10, b=7, c=11, d=13, e=11
        console.log(`a+=1 => ${a+=1}`)                           // a+=1 => 11
        console.log(`a+=1 => ${a+=1}`)                           // a+=1 => 12
        console.log(`b-=2 => ${b-=2}`)                           // b-=2 => 5
        console.log(`b-=2 => ${b-=2}`)                           // b-=2 => 3
        console.log(`c*=2 => ${c*=2}`)                           // c*=2 => 22
        console.log(`c*=2 => ${c*=2}`)                           // c*=2 => 44
        console.log(`d/=2 => ${d/=2}`)                           // d/=2 => 6.5
        console.log(`d/=2 => ${d/=2}`)                           // d/=2 => 3.25
        console.log(`e%=3 => ${e%=3}`)                           // e%=3 => 2
        console.log(`e%=2 => ${e%=2}`)                           // e%=2 => 0
